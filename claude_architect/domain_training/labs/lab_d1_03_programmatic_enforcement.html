Part 2

Programmatic Prerequisite Gates (20 min)

from dataclasses import dataclass, field
from typing import Optional

@dataclass
class AgentState:
    # Tracks verified state — CANNOT be bypassed by the model
    verified_customer_id: Optional[str] = None
    verified_order_id: Optional[str] = None
    tool_call_log: list = field(default_factory=list)


def execute_tool_with_enforcement(tool_name: str, tool_input: dict, state: AgentState) -> dict:
    state.tool_call_log.append(tool_name)

    # GATE 1: lookup_order requires verified customer
    if tool_name == "lookup_order" and state.verified_customer_id is None:
        print("  🚫 BLOCKED: lookup_order — customer not verified")
        return {"isError": True, "errorCategory": "validation", "isRetryable": False,
                "error": "Customer must be verified first.", "required_action": "Call get_customer first."}

    # GATE 2: process_refund requires customer + order
    if tool_name == "process_refund":
        if state.verified_customer_id is None:
            return {"isError": True, "error": "Call get_customer first.", "isRetryable": False}
        if state.verified_order_id is None:
            return {"isError": True, "error": "Call lookup_order first.", "isRetryable": False}

        # GATE 3: Amount threshold
        amount = tool_input.get("amount", 0)
        if amount > 500:
            print(f"  🚫 BLOCKED: ${amount} exceeds $500 threshold")
            return {"isError": True, "errorCategory": "business_rule", "isRetryable": False,
                    "error": f"Refund ${amount} exceeds $500 automated limit.",
                    "required_action": "Call escalate_to_human for manager approval."}

    # Execute the tool (all gates passed)
    if tool_name == "get_customer":
        result = raw_get_customer(tool_input.get("customer_id", ""))
        if "error" not in result:
            state.verified_customer_id = result["customer_id"]
            print(f"  ✅ Customer verified: {result['name']}")
        return result

    elif tool_name == "lookup_order":
        result = raw_lookup_order(tool_input.get("order_id", ""))
        if "error" not in result:
            if result["customer_id"] != state.verified_customer_id:
                return {"isError": True, "error": "Order does not belong to verified customer."}
            state.verified_order_id = result["order_id"]
        return result

    elif tool_name == "process_refund":
        return raw_process_refund(tool_input.get("order_id"), tool_input.get("amount"), tool_input.get("reason"))

    elif tool_name == "escalate_to_human":
        return raw_escalate_to_human(**tool_input)

    return {"error": f"Unknown tool: {tool_name}"}
Part 3

PostToolUse Hook — Data Normalization (15 min)

Raw backend systems return inconsistent formats. The hook normalizes before Claude processes the result — eliminating format reasoning from the model's task.

Raw FormatNormalized Format
account_status_code: 1account_status: "active"
created_unix: 1609459200account_created: "2021-01-01"
order_date: "01/15/2024"order_date: "2024-01-15"
status_code: 3order_status: "delivered"
ACCOUNT_STATUS_MAP = {1: "active", 2: "suspended", 3: "closed"}
ORDER_STATUS_MAP = {1: "placed", 2: "in_transit", 3: "delivered", 4: "cancelled"}

def normalize_tool_result(tool_name: str, result: dict) -> dict:
    if "error" in result: return result  # Don't transform errors

    if tool_name == "get_customer":
        n = result.copy()
        if "created_unix" in n:
            n["account_created"] = datetime.utcfromtimestamp(n.pop("created_unix")).strftime("%Y-%m-%d")
        if "account_status_code" in n:
            n["account_status"] = ACCOUNT_STATUS_MAP.get(n.pop("account_status_code"), "unknown")
        return n

    if tool_name == "lookup_order":
        n = result.copy()
        if "order_date" in n and "/" in str(n["order_date"]):
            n["order_date"] = datetime.strptime(n["order_date"], "%m/%d/%Y").strftime("%Y-%m-%d")
        if "status_code" in n:
            n["order_status"] = ORDER_STATUS_MAP.get(n.pop("status_code"), "unknown")
        return n

    return result

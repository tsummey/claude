Arch

System Architecture

User Query: "Impact of remote work on urban real estate markets" │ ▼ ┌─────────────────────┐ │ COORDINATOR │ ← Decomposes, delegates, aggregates └──────────┬──────────┘ │ (parallel Task calls) ┌──────┴──────┐ │ │ ▼ ▼ ┌────────┐ ┌──────────┐ │SEARCH │ │ DOCUMENT │ ← Specialized, isolated context │AGENT │ │ ANALYSIS │ └────────┘ └──────────┘ │ │ └──────┬──────┘ ▼ ┌─────────────┐ │ SYNTHESIS │ ← Receives EXPLICIT context from coordinator └─────────────┘
Critical Rule — Context Isolation

Subagents have zero access to the coordinator's conversation history. They do not inherit findings automatically. Every piece of context must be passed explicitly in the subagent's prompt. This is the single most common multi-agent bug.

Part 1

Foundation — Mock Data & Subagent Tools (20 min)

import anthropic, json
client = anthropic.Anthropic()

MOCK_SEARCH_RESULTS = {
    "remote work urban real estate": [
        {"title": "Remote Work Exodus", "url": "https://realestate-journal.com/remote-work-exodus",
         "date": "2024-02-15", "excerpt": "Office vacancy rates hit 22% in Q4 2023."},
        {"title": "Suburban Migration Accelerates", "url": "https://urban-studies.org/suburban-migration",
         "date": "2024-01-28", "excerpt": "Suburban home prices up 18% since 2020."}
    ],
}
MOCK_DOCUMENTS = {
    "suburban_housing_study.pdf": {
        "content": "Federal Reserve study: Remote work adds $10,000-$25,000 premium to suburban home values.",
        "date": "2023-11-15", "source": "Federal Reserve Bank of Atlanta"
    },
}

def web_search_agent(query: str, num_results: int = 5) -> dict:
    results = []
    for key, value in MOCK_SEARCH_RESULTS.items():
        if any(word in query.lower() for word in key.split()):
            results.extend(value)
    return {"status": "success", "query": query, "results": results[:num_results]}

def document_analysis_agent(document_name: str, analysis_focus: str) -> dict:
    doc = MOCK_DOCUMENTS.get(document_name)
    if not doc:
        return {"isError": True, "error": f"Document '{document_name}' not found"}
    return {"status": "success", "source": doc["source"], "date": doc["date"],
            "key_findings": doc["content"]}

COORDINATOR_TOOLS = [
    {"name": "web_search_agent", "description": "Search the web for articles and statistics.",
     "input_schema": {"type": "object", "properties": {"query": {"type": "string"}, "num_results": {"type": "integer"}}, "required": ["query"]}},
    {"name": "document_analysis_agent", "description": "Analyze a specific research document in depth.",
     "input_schema": {"type": "object", "properties": {"document_name": {"type": "string"}, "analysis_focus": {"type": "string"}}, "required": ["document_name", "analysis_focus"]}}
]

def execute_subagent(agent_name, agent_input) -> str:
    if agent_name == "web_search_agent":
        result = web_search_agent(agent_input.get("query", ""), agent_input.get("num_results", 5))
    elif agent_name == "document_analysis_agent":
        result = document_analysis_agent(agent_input.get("document_name", ""), agent_input.get("analysis_focus", ""))
    else: result = {"error": "unknown agent"}
    return json.dumps(result, indent=2)
Part 2

The Broken Coordinator — Context Isolation Bug (15 min)

Build this version first. It has the most common multi-agent bug: gathering findings but never passing them to Claude for synthesis.

def run_coordinator_broken(research_query: str) -> str:
    # Phase 1: Gather research
    all_findings = []
    result1 = web_search_agent("remote work urban real estate markets")
    all_findings.append(result1)
    result2 = document_analysis_agent("suburban_housing_study.pdf", "housing values")
    all_findings.append(result2)
    print(f"Gathered {len(all_findings)} sources")

    # Phase 2: Ask Claude to synthesize — BUT WITHOUT PASSING THE FINDINGS
    synthesis_prompt = """Based on your research, synthesize key findings about
    the impact of remote work on urban real estate markets."""
    # BUG: all_findings is NEVER given to Claude

    response = client.messages.create(
        model="claude-opus-4-5", max_tokens=2048,
        messages=[{"role": "user", "content": synthesis_prompt}]
    )
    return response.content[0].text

Part 3

The Correct Coordinator — Explicit Context Passing (25 min)

COORDINATOR_SYSTEM = """You are a research coordinator managing a multi-agent pipeline.

Responsibilities:
1. Decompose the research query into comprehensive subtopics
2. Delegate to specialized agents — do not miss major areas
3. Review ALL findings before requesting synthesis
4. Every factual claim must cite a specific source with date
5. Conflicting data from sources must be noted, not resolved arbitrarily"""

def run_coordinator_correct(research_query: str) -> str:
    messages = [{"role": "user", "content": f"Research comprehensively: {research_query}"}]
    iteration = 0

    while iteration < 15:
        iteration += 1
        response = client.messages.create(
            model="claude-opus-4-5", max_tokens=4096,
            system=COORDINATOR_SYSTEM, tools=COORDINATOR_TOOLS, messages=messages
        )
        print(f"[Iter {iteration}] stop_reason: {response.stop_reason}")

        if response.stop_reason == "end_turn":
            for block in response.content:
                if hasattr(block, "text"): return block.text
            return ""

        elif response.stop_reason == "tool_use":
            messages.append({"role": "assistant", "content": response.content})
            tool_results = []
            for block in response.content:
                if block.type == "tool_use":
                    print(f"  → {block.name}: {json.dumps(block.input)[:80]}")
                    result = execute_subagent(block.name, block.input)
                    tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})
            messages.append({"role": "user", "content": tool_results})
